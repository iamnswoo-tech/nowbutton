📡 앱 → 대시보드 연동 방법

아래 코드 스니펫을 app.js의 _trackEvent 함수 내부에 추가하세요. 측정 완료 시 자동으로 이 대시보드로 이벤트가 전송됩니다.

app.js 수정 코드 (3단계)
// ── YoungButton Analytics SDK ─────────────────
const BEACON_URL = 'https://your-dashboard.vercel.app/api/beacon';
const YB_SID = (() => {
  let s = localStorage.getItem('yb_sid');
  if (!s) { s = 'yb-' + Date.now().toString(36) + '-' + Math.random().toString(36).slice(2,8); localStorage.setItem('yb_sid', s); }
  return s;
})();
function ybBeacon(payload) {
  try {
    const blob = new Blob([JSON.stringify({...payload, sid: YB_SID, t: Date.now()})], {type:'application/json'});
    navigator.sendBeacon ? navigator.sendBeacon(BEACON_URL, blob) : fetch(BEACON_URL,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({...payload,sid:YB_SID,t:Date.now()}),keepalive:true}).catch(()=>{});
  } catch(_) {}
}
// 기존 _trackEvent 함수 내부 맨 끝에 아래 줄 추가
if (name === 'measurement_complete') {
  ybBeacon({ type: 'measurement_complete', category: props.category, score: props.score });
}
if (name === 'app_open') {
  ybBeacon({ type: 'app_open' });
}
if (name === 'page_view') {
  ybBeacon({ type: 'page_view', page: props.page });
}
// vercel.json의 connect-src 항목에 추가:
"connect-src 'self' blob: ... https://your-dashboard.vercel.app"

아래 버튼으로 테스트 이벤트를 전송하고 실시간 탭에서 확인하세요.
